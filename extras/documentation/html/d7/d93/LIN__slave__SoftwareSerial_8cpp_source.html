131void LIN_Slave_SoftwareSerial::handler()
132{
133 // sync frames based on inter-frame pause (not standard compliant!)
134 static uint32_t usLastByte = 0;
135
136 // byte received -> check it
137 if (this->available())
138 {
139 // ESP32 & ESP8266 (BREAK is dropped due to missing stop bit): if SYNC=0x55 received and long time since last byte, start new frame
140 #if defined(ARDUINO_ARCH_ESP32) || defined(ARDUINO_ARCH_ESP8266)
141 if ((this->_serialPeek() == 0x55) && ((micros() - usLastByte) > this->minFramePause))
142 {
143 // set BREAK flag
144 this->flagBreak = true;
145
146 // print debug message
147 DEBUG_PRINT(3, "BRK detected");
148
149 }
150
151 // other architectures (BREAK is received): if BREAK=0x00 received and long time since last byte, start new frame and remove 0x00 from queue
152 #else
153 if ((this->_serialPeek() == 0x00) && ((micros() - usLastByte) > this->minFramePause))
154 {
155 // set BREAK flag
156 this->flagBreak = true;
157
158 // remove 0x00 from FiFo buffer
159 this->_serialRead();
160
161 // print debug message
162 DEBUG_PRINT(3, "BRK detected");
163
164 }
165 #endif
166
167 // store time of this receive
168 usLastByte = micros();
169
170 // call base-class handler
171 LIN_Slave_Base::handler();
172
173 // SoftwareSerial is blocking while sending -> skip reading echo
174 if (this->state == LIN_Slave_Base::STATE_RECEIVING_ECHO)
175 {
176 // propagate to DONE immediately
177 this->state = LIN_Slave_Base::STATE_DONE;
178
179 // optionally disable RS485 transmitter
180 _disableTransmitter();
181 }
182
183 } // if byte received
184
185} // LIN_Slave_SoftwareSerial::handler()
