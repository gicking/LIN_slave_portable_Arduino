140void LIN_Slave_HardwareSerial::handler()
141{
142 // sync frames based on inter-frame pause (not standard compliant!)
143 static uint32_t usLastByte = 0;
144
145 // byte received -> check it
146 if (pSerial->available())
147 {
148 // if 0x00 received and long time since last byte, start new frame
149 if ((pSerial->peek() == 0x00) && ((micros() - usLastByte) > this->minFramePause))
150 this->flagBreak = true;
151
152 // store time of this receive
153 usLastByte = micros();
154
155 // call base-class handler
156 LIN_Slave_Base::handler();
157
158 } // if byte received
159
160} // LIN_Slave_HardwareSerial::handler()
