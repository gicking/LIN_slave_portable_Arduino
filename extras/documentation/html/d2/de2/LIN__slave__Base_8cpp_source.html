30uint8_t LIN_Slave_Base::_calculatePID(uint8_t ID)
31{
32 uint8_t pid; // protected frame ID
33 uint8_t tmp; // temporary variable for calculating parity bits
34
35 // protect ID with parity bits
36 pid = (uint8_t) (ID & 0x3F); // clear upper bits 6 & 7
37 tmp = (uint8_t) ((pid ^ (pid>>1) ^ (pid>>2) ^ (pid>>4)) & 0x01); // pid[6] = PI0 = ID0^ID1^ID2^ID4
38 pid |= (uint8_t) (tmp << 6);
39 tmp = (uint8_t) (~((pid>>1) ^ (pid>>3) ^ (pid>>4) ^ (pid>>5)) & 0x01); // pid[7] = PI1 = ~(ID1^ID3^ID4^ID5)
40 pid |= (uint8_t) (tmp << 7);
41
42 // print debug message (debug level 3)
43 #if defined(LIN_SLAVE_DEBUG_SERIAL) && (LIN_SLAVE_DEBUG_LEVEL >= 3)
44 LIN_SLAVE_DEBUG_SERIAL.print(this->nameLIN);
45 LIN_SLAVE_DEBUG_SERIAL.print(": LIN_Slave_Base::_calculatePID(): PID=0x");
46 LIN_SLAVE_DEBUG_SERIAL.println((int) pid, HEX);
47 #endif
48
49 // return protected ID
50 return pid;
51
52} // LIN_Slave_Base::_calculatePID()

63uint8_t LIN_Slave_Base::_calculateChecksum(uint8_t NumData, uint8_t Data[])
64{
65 uint16_t chk=0x00;
66
67 // LIN2.x uses extended checksum which includes protected ID, i.e. including parity bits
68 // LIN1.x uses classical checksum only over data bytes
69 // Diagnostic frames with ID 0x3C and 0x3D/0x7D always use classical checksum (see LIN spec "2.3.1.5 Checkum")
70 if (!((this->version == LIN_V1) || (pid == 0x3C) || (pid == 0x7D))) // if version 2 & no diagnostic frames (0x3C=60 (PID=0x3C) or 0x3D=61 (PID=0x7D))
71 chk = (uint16_t) this->pid;
72
73 // loop over data bytes
74 for (uint8_t i = 0; i < NumData; i++)
75 {
76 chk += (uint16_t) (Data[i]);
77 if (chk>255)
78 chk -= 255;
79 }
80 chk = (uint8_t)(0xFF - ((uint8_t) chk)); // bitwise invert
81
82 // optional debug output (debug level 3)
83 #if defined(LIN_SLAVE_DEBUG_SERIAL) && (LIN_SLAVE_DEBUG_LEVEL >= 3)
84 LIN_SLAVE_DEBUG_SERIAL.print(this->nameLIN);
85 LIN_SLAVE_DEBUG_SERIAL.print(": LIN_Slave_Base::_calculateChecksum(): CHK=0x");
86 LIN_SLAVE_DEBUG_SERIAL.println((int) chk, HEX);
87 #endif
88
89 // return frame checksum
90 return (uint8_t) chk;
91
92} // LIN_Slave_Base::_calculateChecksum()

135LIN_Slave_Base::LIN_Slave_Base(LIN_Slave_Base::version_t Version, const char NameLIN[], uint32_t TimeoutRx, const int8_t PinTxEN)
136{
137 // Debug serial initialized in begin() -> no debug output here
138
139 // store parameters in class variables
140 this->version = Version; // LIN protocol version (required for checksum)
141 memcpy(this->nameLIN, NameLIN, LIN_SLAVE_BUFLEN_NAME); // node name e.g. for debug
142 this->timeoutRx = TimeoutRx; // timeout [us] for bytes in frame
143 this->pinTxEN = PinTxEN; // optional Tx enable pin for RS485
144
145 // initialize slave node properties
146 this->state = LIN_Slave_Base::STATE_WAIT_FOR_BREAK; // status of LIN state machine
147 this->error = LIN_Slave_Base::NO_ERROR; // last LIN error. Is latched
148 for (uint8_t i=0; i<64; i++)
149 {
150 this->callback[i].type_numData = 0x00; // frame type (high nibble) and number of data bytes (low nibble)
151 this->callback[i].fct = nullptr; // user callback functions (IDs 0x00 - 0x3F)
152 }
153
154 // initialize frame properties
155 this->pid = 0x00; // protected frame identifier
156 this->id = 0x00; // unprotected frame identifier
157 this->numData = 0; // number of data bytes in frame
158 for (uint8_t i=0; i<9; i++)
159 this->bufData[i] = 0x00; // init data bytes (max 8B) + chk
160 this->idxData = 0; // current index in bufData
161 this->timeLastRx = 0; // time [ms] of last received byte in frame
162
163 // initialize TxEN pin low (=transmitter off)
164 if (this->pinTxEN >= 0)
165 {
166 digitalWrite(this->pinTxEN, LOW);
167 pinMode(this->pinTxEN, OUTPUT);
168 }
169
170} // LIN_Slave_Base::LIN_Slave_Base()

301void LIN_Slave_Base::handler()
302{
303 uint8_t chk_calc;
304
305 // on receive timeout [us] within frame reset state machine
306 if (!(this->state | (LIN_Slave_Base::STATE_OFF | LIN_Slave_Base::STATE_WAIT_FOR_BREAK | LIN_Slave_Base::STATE_DONE)) &&
307 ((micros() - this->timeLastRx) > this->timeoutRx))
308 {
309 // set error and abort frame
310 this->error = (LIN_Slave_Base::error_t) ((int) this->error | (int) LIN_Slave_Base::ERROR_TIMEOUT);
311 this->state = LIN_Slave_Base::STATE_DONE;
312
313 // flush receive buffer
314 while (this->available())
315 this->_serialRead();
316
317 // optionally disable RS485 transmitter
318 _disableTransmitter();
319
320 // optional debug output (debug level 1)
321 #if defined(LIN_SLAVE_DEBUG_SERIAL) && (LIN_SLAVE_DEBUG_LEVEL >= 1)
322 LIN_SLAVE_DEBUG_SERIAL.print(this->nameLIN);
323 LIN_SLAVE_DEBUG_SERIAL.print(": LIN_Slave_Base::handler()");
324 LIN_SLAVE_DEBUG_SERIAL.print(": error: frame timeout after ");
325 LIN_SLAVE_DEBUG_SERIAL.print((long) (micros() - this->timeLastRx));
326 LIN_SLAVE_DEBUG_SERIAL.println("us");
327 #endif
328
329 // return immediately
330 return;
331
332 } // if frame receive timeout
333
334
335 // detected LIN BREAK (=0x00 with framing error or inter-frame pause detected)
336 // Note: received BREAK byte is consumed by child class to support also sync on SYNC byte.
337 if (this->_getBreakFlag() == true)
338 {
339 // clear BREAK flag again
340 this->_resetBreakFlag();
341
342 // start frame reception. Note: 0x00 already checked by derived class
343 this->state = LIN_Slave_Base::STATE_WAIT_FOR_SYNC;
344
345 // optionally disable RS485 transmitter
346 _disableTransmitter();
347
348 // optional debug output (debug level 3)
349 #if defined(LIN_SLAVE_DEBUG_SERIAL) && (LIN_SLAVE_DEBUG_LEVEL >= 3)
350 LIN_SLAVE_DEBUG_SERIAL.print(this->nameLIN);
351 LIN_SLAVE_DEBUG_SERIAL.print(": LIN_Slave_Base::handler()");
352 LIN_SLAVE_DEBUG_SERIAL.println(": BREAK detected ");
353 #endif
354
355 } // if BREAK detected
356
357
358 // A byte was received -> handle it
359 if (this->available())
360 {
361 // read received byte and reset timeout timer
362 uint8_t byteReceived = this->_serialRead();
363 this->timeLastRx = micros();
364
365 // optional debug output (debug level 3)
366 #if defined(LIN_SLAVE_DEBUG_SERIAL) && (LIN_SLAVE_DEBUG_LEVEL >= 3)
367 LIN_SLAVE_DEBUG_SERIAL.print(this->nameLIN);
368 LIN_SLAVE_DEBUG_SERIAL.print(": LIN_Slave_Base::handler()");
369 if (this->_getBreakFlag() == true)
370 LIN_SLAVE_DEBUG_SERIAL.print(": BRK, Rx=0x");
371 else
372 LIN_SLAVE_DEBUG_SERIAL.print(": Rx=0x");
373 LIN_SLAVE_DEBUG_SERIAL.println(byteReceived, HEX);
374 #endif
375
376 // handle byte
377 switch (this->state)
378 {
379 // LIN interface disabled, do nothing
380 case LIN_Slave_Base::STATE_OFF:
381 break;
382
383 // just to avoid compiler warning, do nothing. Break state is handled above
384 case LIN_Slave_Base::STATE_WAIT_FOR_BREAK:
385 break;
386
387 // master request frame is finished, do nothing
388 case LIN_Slave_Base::STATE_DONE:
389 break;
390
391 // break has been received, waiting for sync field
392 case LIN_Slave_Base::STATE_WAIT_FOR_SYNC:
393
394 // valid SYNC (=0x55) -> wait for ID
395 if (byteReceived == 0x55)
396 {
397 this->idxData = 0;
398 this->state = LIN_Slave_Base::STATE_WAIT_FOR_PID;
399 }
400
401 // invalid SYNC (!=0x55) -> error
402 else
403 {
404 // set error and abort frame
405 this->error = (LIN_Slave_Base::error_t) ((int) this->error | (int) LIN_Slave_Base::ERROR_SYNC);
406 this->state = LIN_Slave_Base::STATE_DONE;
407
408 // optionally disable RS485 transmitter
409 _disableTransmitter();
410
411 // optional debug output (debug level 1)
412 #if defined(LIN_SLAVE_DEBUG_SERIAL) && (LIN_SLAVE_DEBUG_LEVEL >= 1)
413 LIN_SLAVE_DEBUG_SERIAL.print(this->nameLIN);
414 LIN_SLAVE_DEBUG_SERIAL.print(": LIN_Slave_Base::handler()");
415 LIN_SLAVE_DEBUG_SERIAL.print(": SYNC error, received 0x");
416 LIN_SLAVE_DEBUG_SERIAL.println(byteReceived, HEX);
417 #endif
418
419 } // invalid SYNC
420
421 break; // STATE_WAIT_FOR_SYNC
422
423
424 // sync field has been received, waiting for protected ID
425 case LIN_Slave_Base::STATE_WAIT_FOR_PID:
426
427 this->pid = byteReceived; // received (protected) ID
428 this->id = byteReceived & 0x3F; // extract ID, drop parity bits
429
430 // check PID parity bits 7+8
431 if (this->pid != this->_calculatePID(this->id))
432 {
433 // set error and abort frame
434 this->error = (LIN_Slave_Base::error_t) ((int) this->error | (int) LIN_Slave_Base::ERROR_PID);
435 this->state = LIN_Slave_Base::STATE_DONE;
436
437 // optionally disable RS485 transmitter
438 _disableTransmitter();
439
440 // optional debug output (debug level 1)
441 #if defined(LIN_SLAVE_DEBUG_SERIAL) && (LIN_SLAVE_DEBUG_LEVEL >= 1)
442 LIN_SLAVE_DEBUG_SERIAL.print(this->nameLIN);
443 LIN_SLAVE_DEBUG_SERIAL.print(": LIN_Slave_Base::handler()");
444 LIN_SLAVE_DEBUG_SERIAL.print(": PID parity error, received 0x");
445 LIN_SLAVE_DEBUG_SERIAL.print(this->pid, HEX);
446 LIN_SLAVE_DEBUG_SERIAL.print(", calculated 0x");
447 LIN_SLAVE_DEBUG_SERIAL.println(this->_calculatePID(this->id), HEX);
448 #endif
449
450 } // PID error
451
452 // if slave response ID is registered, call callback function and send response
453 else if ((this->callback[id].fct != nullptr) && (this->callback[id].type_numData & LIN_Slave_Base::SLAVE_RESPONSE))
454 {
455 // get type (high nibble) and number of response bytes (low nibble) from callback array
456 this->type = (LIN_Slave_Base::frame_t) (this->callback[id].type_numData & 0xF0);
457 this->numData = this->callback[id].type_numData & 0x0F;
458
459 // call the user-defined callback function for this ID
460 this->callback[id].fct(numData, this->bufData);
461
462 // attach frame checksum
463 bufData[numData] = this->_calculateChecksum(this->numData, this->bufData);
464
465 // optionally enable RS485 transmitter
466 _enableTransmitter();
467
468 // send slave response (data+chk)
469 this->_serialWrite(bufData, numData+1);
470
471 // advance state to receiving echo
472 this->state = LIN_Slave_Base::STATE_RECEIVING_ECHO;
473
474 // optional debug output (debug level 2)
475 #if defined(LIN_SLAVE_DEBUG_SERIAL) && (LIN_SLAVE_DEBUG_LEVEL >= 2)
476 LIN_SLAVE_DEBUG_SERIAL.print(this->nameLIN);
477 LIN_SLAVE_DEBUG_SERIAL.print(": LIN_Slave_Base::handler()");
478 LIN_SLAVE_DEBUG_SERIAL.print(": handle slave response PID 0x");
479 LIN_SLAVE_DEBUG_SERIAL.println(this->pid, HEX);
480 #endif
481
482 } // if slave response frame
483
484 // if master request ID is registered, get number of data bytes and advance state
485 else if ((this->callback[id].fct != nullptr) && (this->callback[id].type_numData & LIN_Slave_Base::MASTER_REQUEST))
486 {
487 // get type (high nibble) and number of response bytes (low nibble) from callback array
488 this->type = (LIN_Slave_Base::frame_t) (this->callback[id].type_numData & 0xF0);
489 this->numData = this->callback[id].type_numData & 0x0F;
490
491 // advance state to receiving data
492 this->state = LIN_Slave_Base::STATE_RECEIVING_DATA;
493
494 } // if master request frame
495
496 // ID is not registered -> wait for next break
497 else
498 {
499 // optional debug output (debug level 2)
500 #if defined(LIN_SLAVE_DEBUG_SERIAL) && (LIN_SLAVE_DEBUG_LEVEL >= 2)
501 LIN_SLAVE_DEBUG_SERIAL.print(this->nameLIN);
502 LIN_SLAVE_DEBUG_SERIAL.print(": LIN_Slave_Base::handler()");
503 LIN_SLAVE_DEBUG_SERIAL.print(": drop frame PID 0x");
504 LIN_SLAVE_DEBUG_SERIAL.println(this->pid, HEX);
505 #endif
506
507 // reset state machine
508 this->state = LIN_Slave_Base::STATE_WAIT_FOR_BREAK;
509
510 } // if frame not registered
511
512 break; // STATE_WAIT_FOR_PID
513
514
515 // receive master request data
516 case LIN_Slave_Base::STATE_RECEIVING_DATA:
517
518 // check received data
519 this->bufData[(this->idxData)++] = byteReceived;
520
521 // if data is finished, advance to checksum check
522 if (this->idxData >= this->numData)
523 this->state = LIN_Slave_Base::STATE_WAIT_FOR_CHK;
524
525 break; // STATE_RECEIVING_DATA
526
527
528 // receive slave response echo
529 case LIN_Slave_Base::STATE_RECEIVING_ECHO:
530
531 // compare received echo to sent data
532 if (this->bufData[(this->idxData)++] != byteReceived)
533 {
534 // set error and abort frame
535 this->error = (LIN_Slave_Base::error_t) ((int) this->error | (int) LIN_Slave_Base::ERROR_ECHO);
536 this->state = LIN_Slave_Base::STATE_DONE;
537
538 // optionally disable RS485 transmitter
539 _disableTransmitter();
540
541 // optional debug output (debug level 1)
542 #if defined(LIN_SLAVE_DEBUG_SERIAL) && (LIN_SLAVE_DEBUG_LEVEL >= 1)
543 LIN_SLAVE_DEBUG_SERIAL.print(this->nameLIN);
544 LIN_SLAVE_DEBUG_SERIAL.print(": LIN_Slave_Base::handler()");
545 LIN_SLAVE_DEBUG_SERIAL.print(": echo error, received 0x");
546 LIN_SLAVE_DEBUG_SERIAL.print(byteReceived, HEX);
547 LIN_SLAVE_DEBUG_SERIAL.print(", expected 0x");
548 LIN_SLAVE_DEBUG_SERIAL.println(this->bufData[(this->idxData)-1], HEX);
549 #endif
550
551 } // if echo error
552
553 // if data is finished, finish frame
554 else if (this->idxData >= this->numData+1)
555 {
556 this->state = LIN_Slave_Base::STATE_DONE;
557
558 // optionally disable RS485 transmitter
559 _disableTransmitter();
560 }
561
562 break; // STATE_RECEIVING_ECHO
563
564
565 // Data has been received for master request frame, waiting for checksum
566 case LIN_Slave_Base::STATE_WAIT_FOR_CHK:
567
568 // calculate checksum for master request frame
569 chk_calc = this->_calculateChecksum(this->numData, this->bufData);
570
571 // Checksum valid -> call user-defined callback function for this ID
572 if (byteReceived == chk_calc)
573 {
574 // call user-defined master request callback function. Only reachable if callback has been registered
575 this->callback[id].fct(numData, bufData);
576
577 // optional debug output (debug level 2)
578 #if defined(LIN_SLAVE_DEBUG_SERIAL) && (LIN_SLAVE_DEBUG_LEVEL >= 2)
579 LIN_SLAVE_DEBUG_SERIAL.print(this->nameLIN);
580 LIN_SLAVE_DEBUG_SERIAL.print(": LIN_Slave_Base::handler()");
581 LIN_SLAVE_DEBUG_SERIAL.print(": handle master request PID 0x");
582 LIN_SLAVE_DEBUG_SERIAL.println(this->pid, HEX);
583 #endif
584
585 } // if checksum ok
586
587 // checksum error
588 else
589 {
590 // set error
591 this->error = (LIN_Slave_Base::error_t) ((int) this->error | (int) LIN_Slave_Base::ERROR_CHK);
592
593 // optional debug output (debug level 1)
594 #if defined(LIN_SLAVE_DEBUG_SERIAL) && (LIN_SLAVE_DEBUG_LEVEL >= 1)
595 LIN_SLAVE_DEBUG_SERIAL.print(this->nameLIN);
596 LIN_SLAVE_DEBUG_SERIAL.print(": LIN_Slave_Base::handler()");
597 LIN_SLAVE_DEBUG_SERIAL.print(": CHK error, received 0x");
598 LIN_SLAVE_DEBUG_SERIAL.print(byteReceived, HEX);
599 LIN_SLAVE_DEBUG_SERIAL.print(", calculated 0x");
600 LIN_SLAVE_DEBUG_SERIAL.println(chk_calc, HEX);
601 #endif
602
603 } // if checksum error
604
605 // frame is finished
606 this->state = LIN_Slave_Base::STATE_DONE;
607
608 // optionally disable RS485 transmitter
609 _disableTransmitter();
610
611 break; // STATE_WAIT_FOR_CHK
612
613
614 // this should never happen -> error
615 default:
616
617 // set error and abort frame
618 this->error = (LIN_Slave_Base::error_t) ((int) this->error | (int) LIN_Slave_Base::LIN_Slave_Base::ERROR_STATE);
619 this->state = LIN_Slave_Base::STATE_DONE;
620
621 // optionally disable RS485 transmitter
622 _disableTransmitter();
623
624 // optional debug output (debug level 1)
625 #if defined(LIN_SLAVE_DEBUG_SERIAL) && (LIN_SLAVE_DEBUG_LEVEL >= 1)
626 LIN_SLAVE_DEBUG_SERIAL.print(this->nameLIN);
627 LIN_SLAVE_DEBUG_SERIAL.print(": LIN_Slave_Base::handler()");
628 LIN_SLAVE_DEBUG_SERIAL.print(": error: illegal state ");
629 LIN_SLAVE_DEBUG_SERIAL.print(this->state);
630 LIN_SLAVE_DEBUG_SERIAL.println(", this should never happen...");
631 #endif
632
633 } // switch(state)
634
635 } // if byte received
